analyzing jvm logs (garbage collector pause)
awk '/Pause/ {
    split($0, a, "]");
    timestamp = substr(a[1], 2);
    duration = $NF;
    gsub(/ms/, "", duration);
    print duration " " timestamp
}' gc-*.log | sort -nr | head -n 10
